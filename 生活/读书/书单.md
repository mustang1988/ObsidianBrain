
```dataviewjs
const status = [ "阅读中", "未读" ];
dv.table(
	["", "书名", "文体", "作者", "简介", "标签", "阅读状态"],
	dv.pages('"生活/读书/书库"')
	.sort(book => [ status.indexOf(book.Status), book.Nation, book.Series, book.NoInSeries, book.file.name ], 'asc')
	.map(book => {
		const author = `[${book.Nation.substring(0,1)}]${book.Author}`;
		const abstract = book.Abstract.length > 20 
			? `${book.Abstract.substring(0, 20)}...`
			: book.Abstract;
		const link = book.file.link;
		link.display = book.Series 
			? `${book.Series}(${book.NoInSeries}): ${book.file.name.substring(book.file.name.indexOf("：")+1)}`
			: undefined;
		return [
			book.Cover,
			book.file.link,
			book.Type,
			author,
			abstract,
			book.file.etags.join('<br/>'),
			book.Status
		]
	})
);
```
